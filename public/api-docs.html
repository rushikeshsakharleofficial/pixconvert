PixConvert API

Open-source PDF & image processing API. No authentication required.

Base URL

/api/v1

Rate Limit

10 req/sec

Max File Size

50 MB

Auth

None (open)

Quick Start

# Compress a PDF
curl -X POST http://localhost:4000/api/v1/compress-pdf \
  -F "files=@document.pdf" -o compressed.pdf

# Merge two PDFs
curl -X POST http://localhost:4000/api/v1/merge-pdf \
  -F "files=@doc1.pdf" -F "files=@doc2.pdf" -o merged.pdf

# Convert from URL
curl -X POST http://localhost:4000/api/v1/compress-pdf \
  -H "Content-Type: application/json" \
  -d '{"urls":["https://example.com/doc.pdf"]}' -o result.pdf

# Get download URL instead of binary
curl -X POST "http://localhost:4000/api/v1/compress-pdf?output=url" \
  -F "files=@document.pdf"

Input Methods

# 1. File upload (multipart/form-data)
-F "files=@local-file.pdf"

# 2. URL input (JSON body)
-H "Content-Type: application/json" -d '{"urls":["https://..."]}'

# 3. Mixed (upload + URL)
-F "files=@local.pdf" -F "urls=https://example.com/remote.pdf"

Output Modes

# Default: binary stream (pipe to file)
curl ... -o result.pdf

# URL mode: returns JSON with download link
curl ... "?output=url"
# → {"success":true,"url":"/downloads/abc123.pdf","filename":"compressed.pdf","size":42000,"expiresAt":"..."}

Endpoints

Error Responses

// 400 Bad Request
{"success": false, "error": "At least 1 file(s) required. Got 0."}

// 413 Payload Too Large
{"success": false, "error": "File too large. Max 50MB allowed."}

// 429 Rate Limited
{"success": false, "error": "Rate limit exceeded. Max 10 requests per second."}

// 502 Processing Error
{"success": false, "error": "Ghostscript failed: ..."}
